	CBLOCK	0x20	;ENDEREÃ‡O INICIAL DA MEMÃ“RIA DE
					;USUÃRIO
		W_TEMP		;REGISTRADORES TEMPORÃRIOS PARA USO
		STATUS_TEMP	;JUNTO Ã€S INTERRUPÃ‡Ã•ES
		;COLOQUE AQUI SUAS NOVAS VARIÃVEIS
		;NÃƒO ESQUEÃ‡A COMENTÃRIOS ESCLARECEDORES

;* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
;*                    INÃCIO DA INTERRUPÃ‡ÃƒO                        *
;* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
; ENDEREÃ‡O DE DESVIO DAS INTERRUPÃ‡Ã•ES. A PRIMEIRA TAREFA Ã‰ SALVAR OS
; VALORES DE "W" E "STATUS" PARA RECUPERAÃ‡ÃƒO FUTURA

	ORG	0x04			;ENDEREÃ‡O INICIAL DA INTERRUPÃ‡ÃƒO
	MOVWF	W_TEMP		;COPIA W PARA W_TEMP
	SWAPF	STATUS,W

;* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
;*                    ROTINA DE INTERRUPÃ‡ÃƒO                        *
;* * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * *
; AQUI SERÃƒO ESCRITAS AS ROTINAS DE RECONHECIMENTO E TRATAMENTO DAS
; INTERRUPÃ‡Ã•ES'

	;TESTA O BIT 5
	BTFSS GPIO, 5 
	GOTO  0TESTA_B4
	GOTO  100TESTA_B1
	
	0TESTA_B4:
	BTFSS GPIO, GP4
	GOTO 00TESTA_B2
	GOTO 01TESTA_B2
	
	00TESTA_B2:
	BTFSS GPIO, GP2
	GOTO 000TESTA_B1
	GOTO 001TESTA_B1
	
	01TESTA_B2:
	BTFSS GPIO, GP2
	GOTO 010TESTA_B1
	GOTO 011TESTA_B1
	
	000TESTA_B1:
	BTFSS GPIO, GP1
	GOTO R0000
	GOTO R0001
	
	001TESTA_B1:
	BTFSS GPIO, GP1
	GOTO R0010
	GOTO R0011
	
	010TESTA_B1:
	BTFSS GPIO, GP1
	GOTO R0100
	GOTO R0101
	
	011TESTA_B1:
	GOTO R0110
	GOTO R0111
	
	100TESTA_B1:
	GOTO R1000
	GOTO R1001
		
	R0000:
	BSF GPIO, GP1
	GOTO SAI_INT
	
	R0001:
	BSF GPIO, GP2
	BCF GPIO, GP1
	GOTO SAI_INT
	
	R0010:
	BSF GPIO, GP1
	GOTO SAI_INT

	R0011:
	BCF GPIO, GP1
	BCF GPIO, GP2
	BSF GPIO, GP4
	GOTO SAI_INT
	
	R0100:
	BSF GPIO, GP1
	GOTO SAI_INT
	
	R0101:
	BCF GPIO, GP1
	BSF GPIO, GP2
	GOTO SAI_INT
	
	R0110:
    	BSF GPIO, GP1
	GOTO SAI_INT
	
	R0111:
	BCF GPIO, GP1
	BCF GPIO, GP2
	BCF GPIO, GP4
	BSF GPIO, GP5
	GOTO SAI_INT
	
	R1000:
	BSF GPIO, GP1
	GOTO SAI_INT
	
	R1001:
	BCF GPIO, GP5
	BTFSS GPIO, GP0
	GOTO ACENDE
	GOTO APAGA

	ACENDE:
	BSF GPIO, GP0
	GOTO SAI_INT

	APAGA:
	BCF GPIO, GP0
	GOTO SAI_INT

;*                 ROTINA DE SAÃDA DA INTERRUPÃ‡ÃƒO                  *

	SWAPF	W_TEMP,W	;MOVE W_TEMP PARA W
	RETFIE
